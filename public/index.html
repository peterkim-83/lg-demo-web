SELECT SUM(CASE WHEN rn_asc = 1 THEN amount ELSE 0 END) AS first_snapshot_open_pipeline_amount,
SUM(CASE WHEN rn_desc = 1 THEN amount ELSE 0 END) AS latest_snapshot_open_pipeline_amount,
